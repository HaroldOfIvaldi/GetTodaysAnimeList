/* ----- Get today's Link or specified day ----- */ /* ----- Variables ----- */ const daysOfWeek = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday']; const separator = '&tag='; const today = daysOfWeek[new Date().getDay()]; /* ----- Get URL Input Variables ----- */ const urlParams = new URLSearchParams(window.location.search); const baseWebPage = urlParams.get('baseWebPage'); const day = urlParams.get('day') || today; /* ----- Run Script ----- */ const webPage = baseWebPage + separator + day; window.open(webPage, '_blank');
